package com.elad.koshercalendar.com.elad.com.elad.koshercalendar

import com.kosherjava.zmanim.ComplexZmanimCalendar
import com.kosherjava.zmanim.util.GeoLocation
import java.time.*

data class ZmanResults(
    val netz: LocalTime?,
    val chatzot: LocalTime?,
    val minchaKetana: LocalTime?,
    val shkia: LocalTime?,
    val tzeit: LocalTime?
)

object ZmanimProvider {
    fun compute(date: LocalDate, lat: Double, lon: Double, tz: ZoneId): ZmanResults {
        val cal = ComplexZmanimCalendar(
            GeoLocation("Hod Hasharon", lat, lon, 0.0, TimeZone.getTimeZone(tz))
        )
        cal.gCalendar.set(date.year, date.monthValue - 1, date.dayOfMonth)

        DebugLog.d("Computing zmanim for $date lat=$lat lon=$lon tz=$tz")

        return ZmanResults(
            netz = cal.sunrise?.toInstant()?.atZone(tz)?.toLocalTime(),
            chatzot = cal.chatzos?.toInstant()?.atZone(tz)?.toLocalTime(),
            minchaKetana = cal.minchaKetana?.toInstant()?.atZone(tz)?.toLocalTime(),
            shkia = cal.sunset?.toInstant()?.atZone(tz)?.toLocalTime(),
            tzeit = cal.tzais?.toInstant()?.atZone(tz)?.toLocalTime()
        )
    }
}
